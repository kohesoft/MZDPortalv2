<!--
To allow file uploads up to 10MB, add the following configuration to your web.config:

In the <system.web> section:
-->
<httpRuntime maxRequestLength="10240" executionTimeout="3600" />

<!--
In the <system.webServer> section:
-->
<security>
  <requestFiltering>
    <requestLimits maxAllowedContentLength="10485760" />
  </requestFiltering>
</security>
